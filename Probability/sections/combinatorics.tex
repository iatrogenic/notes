\begin{definition} 
    Let $n$ be a natural number. The \emph{factorial} of $n$, denoted $n!$, is defined by
    \begin{equation*}
        \begin{cases}
            1 & \text{if } n = 0, \\
            n(n-1)! & \text{otherwise}.
        \end{cases}
    \end{equation*}
\end{definition}

\begin{definition}
    Let $n \geq k \geq 0$ positive integers. The binomial coefficient is defined as
    \begin{equation}
        \binom{n}{k} \coloneqq \frac{n!}{k!(n-k)!}
    \end{equation}  
\end{definition}
